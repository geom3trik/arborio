Entity: struct
  entity_name: string
  hitboxes: List[Rect]
  standard_draw: EntityDraw
  selected_draw: EntityDraw
  minimum_size_x: int
  minimum_size_y: int
  resizable_x: bool
  resizable_y: bool
  attribute_info: List[AttributeInfo]

AttributeInfo: struct
  name: string
  type: TypeString | TypeFloat | TypeInt | TypeBool
  options: Optional[List[string] | List[float] | List[int]]

EntityDraw: struct
  initial_draw: DrawList
  node_draw: DrawList

DrawList: List[DrawElement]

DrawElement: DrawImage | DrawRect | DrawLine | DrawCurve | DrawTiles | Draw9Slice

DrawImage: struct
  sprite: Expression
  scale: Vec2
  bounds: Rect

DrawTiles: struct
  tileset: Expression  # this can be either a texture path or a tileset ID
  bounds: Rect
  tiler: AutotilerType

AutotilerType: enum
  fg
  bg
  cassette

# for 9slice drawings that are larger than 8*3 pixels square, treat the middles as tiled
Draw9Slice: struct
  sprite: Expression
  bounds: Rect

DrawLine: struct
  start: Vec2
  end: Vec2
  color: Color
  arrowhead: bool
  thickness: int

DrawCurve: struct
  start: Vec2
  end: Vec2
  middle: Vec2
  color: Color
  thickness: int

DrawRect: struct
  rect: Rect
  color: Color
  color_border: Color

Vec2: struct
  x: Expression
  y: Expression

Color: struct
  r: Expression
  g: Expression
  b: Expression
  a: Expression

Rect:
  topleft: Vec2
  size: Vec2

Expression: Add | Sub | Mul | Div | Mod | Neg | Match | Atom | Const

Add: struct
  lhs: Expression
  rhs: Expression
Sub: struct
  lhs: Expression
  rhs: Expression
Mul: struct
  lhs: Expression
  rhs: Expression
Div: struct
  lhs: Expression
  rhs: Expression
Mod: struct
  lhs: Expression
  rhs: Expression
Neg: struct
  expr: Expression
Match: struct
  arms: List[MatchArm]
  default: Expression
MatchArm: struct
  match: Const
  value: Expression
Atom: string  # the name of the attribute to load the value of as part of this expression
              # the following automatic attributes will also be provided:
              # x, y, width, height, nodex, nodey, nextnodex, nextnodey, nextnodexorbase, nextnodeyorbase, prevnodex, prevnodey, prevnodexorbase, prevnodeyorbase, firstnodex, firstnodey, lastnodex, lastnodey
Const: float | string
